#!/bin/bash
# location /scripts/function.sh

# The generic function
general() {
    hosts
    dns_cfg
    close_selinux
    time_zone
    time_sync
    login_prompt
    command
    swap_use
    open_files
    install_package
}

CentOS6_optimization() {
    close_6_firewall
    close_6_NetworkManager
    6_lan_code
    6_ssh_cfg
    6_alias
    CentOS6_local_yum
}

CentOS7_optimization() {
    close_7_firewall
    close_7_NetworkManager
    7_lan_code
    7_ssh_cfg
    7_alias
    CentOS7_local_yum
    modify_network_name
}


# Write the hostname to hosts
hosts() {
    ip_addr=$(hostname -I | awk '{print $1}')
    echo "$ip_addr `hostname`" >>/etc/hosts
}

# Writes the DNS address to /etc/resolv.conf
dns_cfg() {
cat >> /etc/resolv.conf <<EOF
nameserver 223.5.5.5
nameserver 114.114.114.114
EOF
}

# Disable selinux
close_selinux() {
    setenforce 0
    sed -i '/^SELINUX=/{s#.*#SELINUX=disabled#}' /etc/selinux/config
}

# Close the firewall
close_7_firewall() {
    systemctl stop firewalld
    systemctl disable firewalld &>/dev/null
}

close_6_firewall() {
    /etc/init.d/iptables stop &>/dev/null
    chkconfig iptables off
}

# Close NetworkManager
close_7_NetworkManager() {
    systemctl stop NetworkManager &>/dev/null
    systemctl disable NetworkManager &>/dev/null
}

close_6_NetworkManager() {
    /etc/init.d/NetworkManager stop &>/dev/null
    chkconfig NetworkManager off &>/dev/null
}

# Set the time zone and time_sync
time_zone() {
    ln -sf /usr/share/zoneinfo/Asia/Shanghai /etc/localtime
}

time_sync() {
    hwclock -w --localtime
    echo "*/10 * * * * /usr/sbin/ntpdate ntp1.aliyun.com &>/dev/null; /sbin/hwclock -w" | crontab
}

# Set the language and encoding
6_lan_code() {
   sed -i '/^LANG/ c LANG=en_US.UTF-8' /etc/sysconfig/i18n
}

7_lan_code() {
    localectl set-locale LANG=en_US.UTF-8
}

# SSH optimization
6_ssh_cfg() {
    sed -i '/^GSSAPIAuthentication/s/yes/no/' /etc/ssh/sshd_config
    sed -i '/^#UseDNS/c UseDNS no' /etc/ssh/sshd_config
    /etc/init.d/sshd reload &>/dev/null
}

7_ssh_cfg() {
    sed -i '/^GSSAPIAuthentication/s/yes/no/' /etc/ssh/sshd_config
    sed -i '/^#UseDNS/c UseDNS no' /etc/ssh/sshd_config
    systemctl reload sshd 
}

# Turn off the login prompt
login_prompt() {
    > /etc/issue
    > /etc/issue.net
}

# Displays the history command time and Command Prompt
command() {
    echo 'export HISTTIMEFORMAT="%F %T `whoami` --> "' >>/etc/profile
    echo 'PS1="\[\e[34;40m\][\u@\h \t \w]\\$ \[\e[0m\]"' > /etc/profile.d/PS1.sh
}


# The alias optimization
6_alias() {
cat >>/etc/profile.d/alias.sh<<EOF
alias grep='grep --color=auto' 2>/dev/null
alias egrep='egrep --color=auto' 2>/dev/null
alias fgrep='fgrep --color=auto' 2>/dev/null
alias vi='vim' 2>/dev/null
EOF
}

7_alias() {
cat >>/etc/profile.d/alias.sh<<EOF
alias vi='vim' 2>/dev/null
EOF
}

# Reduce the use of swap
swap_use() {
    echo "10" > /proc/sys/vm/swappiness
    echo "vm.swappiness=10" >>/etc/sysctl.conf
}

# Sets the maximum number of open files
open_files() {
    ulimit -n 65535
cat >> /etc/security/limits.conf <<EOF
* soft nofile 65535
* hard nofile 65535
EOF
}

# Use local yum source
CentOS6_local_yum()
{
if [ -b /dev/sr1 ]; then
mkdir /media/cdrom{1,2}
echo "/dev/sr0 /media/cdrom1 iso9660 defaults 0 0" >>/etc/fstab
echo "/dev/sr1 /media/cdrom2 iso9660 defaults 0 0" >>/etc/fstab

cd /etc/yum.repos.d/ && mkdir repo_bak && mv *.repo repo_bak

cat >>/etc/yum.repos.d/CentOS6-local.repo <<EOF
[CentOS6-local]
name=CentOS6
baseurl=file:///media/cdrom1
        file:///media/cdrom2
enabled=1
gpgcheck=0
EOF

else

echo "/dev/sr0 /media iso9660 defaults 0 0" >>/etc/fstab
cd /etc/yum.repos.d/ && mkdir repo_bak && mv *.repo repo_bak
cat >>/etc/yum.repos.d/CentOS6-local.repo <<EOF
[CentOS6-local]
name=CentOS6
baseurl=file:///media/
enabled=1
gpgcheck=0
EOF
fi

mount -a &>/dev/null
yum clean all &>/dev/null
yum makecache &>/dev/null
}

CentOS7_local_yum()
{
echo "/dev/sr0 /media iso9660 defaults 0 0" >>/etc/fstab
mount -a &>/dev/null

cd /etc/yum.repos.d/ && mkdir repo_bak && mv *.repo repo_bak

cat >>/etc/yum.repos.d/CentOS7-local.repo <<EOF
[CentOS7-local]
name=CentOS7
baseurl=file:///media/
enabled=1
gpgcheck=0
EOF

yum clean all &>/dev/null
yum makecache &>/dev/null
}

# Use aliyun's source 
aliyun_source() {
cd /etc/yum.repos.d/ && mkdir repo_bak && mv *.repo repo_bak

if grep " 7\.[0-9]" /etc/centos-release &>/dev/null; then
    curl -o /etc/yum.repos.d/epel.repo http://mirrors.aliyun.com/repo/epel-7.repo &>/dev/null
    curl -o /etc/yum.repos.d/CentOS-Base.repo http://mirrors.aliyun.com/repo/Centos-7.repo &>/dev/null
elif grep " 6\.[0-9]" /etc/centos-release &>/dev/null; then
    curl -o /etc/yum.repos.d/epel.repo http://mirrors.aliyun.com/repo/epel-6.repo &>/dev/null
    curl -o /etc/yum.repos.d/CentOS-Base.repo http://mirrors.aliyun.com/repo/Centos-6.repo &>/dev/null
fi

yum clean all &>/dev/null
yum makecache &>/dev/null
}

# Installation maintenance installation package
install_package() {
for i in gcc gcc-c++ vim-enhanced sysstat net-tools wget ntpdate lsof bind-utils automake autoconf lrzsz cmake nmap dos2unix tree
do
    if ! rpm -q $i &>/dev/null; then
        yum -y install $i &>/dev/null
    fi
done
}

# Modify the network interface name for CentOS7_ens33
modify_network_name() {
if ip a | grep ens33 &>/dev/null; then
    cp /etc/default/grub /etc/default/grub.bak
    cp /boot/grub2/grub.cfg /boot/grub2/grub.cfg.bak
    sed -i '/GRUB_CMDLINE_LINUX/c GRUB_CMDLINE_LINUX="rhgb quiet net.ifnames=0 biosdevname=0"' /etc/default/grub
    cd /etc/sysconfig/network-scripts/ && mv ifcfg-ens33 ifcfg-eth0
    sed -i '/DEVICE/c DEVICE=eth0' ifcfg-eth0
    sed -i '/NAME/c NAME=eth0' ifcfg-eth0
    grub2-mkconfig -o /boot/grub2/grub.cfg &>/dev/null
fi
}
