#!/bin/bash

ROOT_UID=0
E_NOTROOT=85

if [ "$UID" -ne "$ROOT_UID" ]
then
  echo "Must be root to run this script."
  exit $E_NOTROOT
fi

test_file=/tmp/test_report.txt

source /etc/profile.d/PS1.sh
[[ -n `echo $PS1 | grep "34;40m"` ]] || echo "PS1 set wrong!" >> $test_file
grep `hostname` /etc/hosts &>/dev/null || echo "hosts wrong!" >> $test_file
dig +short www.baidu.com &>/dev/null || echo "dns set wrong!" >> $test_file

[[ `getenforce` == "Disabled" || `getenforce` == "Permissive" ]] || echo "selinux wrong!" >> $test_file

file /etc/localtime | grep "Asia/Shanghai" &>>/dev/null || echo "time_zone wrong!" >> $test_file

hwclock --debug | grep "on local time" &>/dev/null || echo "hwclock is not use localtime!" >> $test_file

crontab -l | grep ntpdate &>/dev/null || echo "ntpdate cron task error!" >> $test_file

[[ -z `cat /etc/issue` ]] || echo "/etc/issue is not empty!" >> $test_file
[[ -z `cat /etc/issue.net` ]] || echo "/etc/issue.net is not empty!" >> $test_file

[[ $HISTTIMEFORMAT == "%F %T root --> " ]] || echo "history format error!" >> $test_file

[[ `cat /proc/sys/vm/swappiness` -eq 10 ]] || echo "swap use error!" >> $test_file

[[ `ulimit -Sn` -eq 65535 && `ulimit -Hn` -eq 65535 ]] || echo "open files set error!" >> $test_file

for i in gcc gcc-c++ vim-enhanced sysstat net-tools wget ntpdate lsof bind-utils automake autoconf lrzsz cmake nmap dos2unix tree
do
    if ! rpm -q $i &>/dev/null; then
        echo " package --->$i<--- is not installed! " >> $test_file
    fi
done

[[ $LANG == "en_US.UTF-8" ]] || echo "LANG set error!" >> $test_file

[[ `awk '/^GSSAPIAuthentication/{print $2}' /etc/ssh/sshd_config` == no ]] || echo "sshcfg_GSSAPI set error!" >> $test_file

[[ `awk '/UseDNS/{print $2}' /etc/ssh/sshd_config` == no ]] || echo "sshcfg_UseDNS set error!" >> $test_file

if ! yum repolist | grep "CentOS[67]-local" &>/dev/null; then
 echo "local source error" >> $test_file
fi

if grep " 6\.[0-9]" /etc/centos-release &>/dev/null; then
    if ! egrep "vi|egrep|fgrep|grep" /etc/profile.d/alias.sh &>/dev/null; then
        echo "alias set error!" >> $test_file
    fi
/etc/init.d/iptables status &>/dev/null && echo "iptables is active,set wrong!" >> $test_file
[[ -z `chkconfig --list iptables | grep on` ]] || echo "iptables is enabled,set wrong!" >> $test_file
    if rpm -q NetworkManager &>/dev/null; then
    service NetworkManager status &>/dev/null && echo "NetworkManager is active,set wrong!" >> $test_file
    [[ -z `chkconfig --list NetworkManager | grep on` ]] || echo "NetworkManager is enabled,set wrong!" >> $test_file
    fi
elif grep " 7\.[0-9]" /etc/centos-release &>/dev/null; then
    if ! grep "vi" /etc/profile.d/alias.sh &>/dev/null; then
        echo "alias set error!" >> $test_file
    fi
[[ `systemctl status NetworkManager firewalld | grep "inactive (dead)" | wc -l` -eq 2 ]] || echo "NetworkManager or firewalld is alive, set is wrong!" >> $test_file
[[ `systemctl status NetworkManager firewalld | grep " disabled;" | wc -l` -eq 2 ]] || echo "NetworkManager or firewalld is enabled, set is wrong!" >> $test_file
if ! ip a s eth0 &>/dev/null; then echo "The network card name was not modified successfully" >> $test_file; fi
    if rpm -q iptables-services &>/dev/null; then
    systemctl status iptables &>/dev/null && echo "iptables is active,set wrong!" >> $test_file
    [[ `systemctl status iptables | grep " disabled;" | wc -l` -eq 1 ]] || echo "iptables is enabled,set wrong!" >> $test_file
    fi
fi

[[ -s $test_file ]] && echo "------> something wrong! Please check $test_file! <------" || echo "------> Successful initialization! <------"
