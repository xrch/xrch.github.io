#!/bin/bash

ROOT_UID=0
E_NOTROOT=85

if [ "$UID" -ne "$ROOT_UID" ]
then
  echo "Must be root to run this script."
  exit $E_NOTROOT
fi

export PATH=/sbin:/bin:/usr/sbin:/usr/bin

source /scripts/function.sh

if grep " 6\.[0-9]" /etc/centos-release &>/dev/null; then
    CentOS6_optimization
elif grep " 7\.[0-9]" /etc/centos-release &>/dev/null; then
    CentOS7_optimization
    yum -y install iptables-services &>/dev/null
    systemctl stop iptables.service &>/dev/null
    systemctl disable iptables.service &>/dev/null
fi

general

# Completed task
echo "     ----->After initialization, you should restart system.<-----     "
exit 0
